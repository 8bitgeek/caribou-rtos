136 /**
137  * @brief Calculate the current fill level of the queue.
138  * @param queue Pointer to a previously initialized caribou_bytequeue_t structure.
139  * @return The queue fill-level expressed as the difference in number of bytes between the head and tail pointers.
140  */
141 int caribou_bytequeue_level(caribou_bytequeue_t* queue)
142 {
143  int state = caribou_interrupts_disable();
144  int rc = queue->head - queue->tail;
145  if ( rc < 0 )

147  rc = (queue->size - queue->tail) + queue->head;
148  }
149  caribou_interrupts_set(state);

173 /**
174  * @brief Put a byte into the queue.
175  * @param queue Pointer to a previously initialized caribou_bytequeue_t structure.
176  * @param c The byte to place on the queue.
177  * @return Return true if the byte was stored, else false (queue full).
178  */
179 bool caribou_bytequeue_put(caribou_bytequeue_t* queue,uint8_t c)
180 {
181  int state = caribou_interrupts_disable();
182  bool rc = !_queue_full(queue); // not full?
183  if ( rc )
184  {
185  queue->queue[queue->head++] = c;
186  if ( queue->head >= queue->size )
187  queue->head = 0;
188  }
189  caribou_interrupts_set(state);
190  return rc;
191 }

193 /**
194  * @brief Put a sequence of bytes into the queue.
195  * @param queue Pointer to a previously initialized caribou_bytequeue_t structure.
196  * @param buf The buffer from where the bytes will be retrieved.
197  * @param sz The maximum number of bytes to put from the buffer, buf must be at least this large.
198  * @return The number of bytes stored may not equal the input sz if the queue was filled before all bytes stored.
199  */
200 int caribou_bytequeue_puts(caribou_bytequeue_t* queue,uint8_t* c,int sz)
201 {
202  int rc=0;
203  int state = caribou_interrupts_disable();
204  while( rc<sz && !_queue_full(queue) )
205  {
206  queue->queue[queue->head++] = c[rc++];
207  if ( queue->head >= queue->size )
208  queue->head = 0;
209  }
210  caribou_interrupts_set(state);
211  return rc;
