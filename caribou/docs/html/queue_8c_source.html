22 /** **************************************************************************
23  ** @brief Allocate and initialize a new queue on the stack
24  ** @param depth The depth of the queue expressed in the number of pointers.
25  ** @return A pointer to the new queue or NULL if allocation failed.
26  *****************************************************************************/
27 caribou_queue_t* caribou_queue_new(int depth)
28 {
29  caribou_queue_t* queue=(caribou_queue_t*)malloc(sizeof(caribou_queue_t));
30  if(queue)
31  {
32  if ( depth == QUEUE_DEPTH_DYNAMIC )
33  {
34  queue->msgs=(caribou_queue_msg_t**)NULL;
35  queue->depth = depth;
36  queue->count=0;
37  }
38  else
39  {
40  queue->msgs=(caribou_queue_msg_t**)malloc(depth*sizeof(caribou_queue_msg_t*));
41  if ( queue->msgs )
42  {
43  memset(&queue->msgs[0],0,depth*sizeof(caribou_queue_msg_t*));
44  queue->depth = depth;
45  queue->count=0;
46  }
47  else
48  {
49  free(queue);
50  queue=NULL;
51  }
52  }
53  }
54  return queue;
55 }

220 /** **************************************************************************
221  ** @brief Post a message to the queue and don't block if there is no space in
222  ** the queue.
223  ** @param queue The queue to operate on.
224  ** @param msg he message to insert.
225  ** @return true if message was posted
226  *****************************************************************************/
227 bool caribou_queue_try_post(caribou_queue_t* queue, caribou_queue_msg_t* msg)
228 {
229  bool rc=false;
230  int lvl = caribou_queue_lock();
231  if ( queue->depth == QUEUE_DEPTH_DYNAMIC ) // dynamic queue?
232  {
233  queue->msgs = (caribou_queue_msg_t**)realloc(queue->msgs,(queue->count+1)*sizeof(caribou_queue_msg_t*));
234  if ( queue->msgs )
235  {
236  queue->msgs[queue->count++] = msg;
237  rc=true;
238  }
239  }
240  else if ( queue->count < queue->depth ) // static queue has space?
241  {
242  queue->msgs[queue->count++] = msg;
243  rc=true;
244  }
245  caribou_queue_lock_restore(lvl);
246  return rc;
