178 /**
179  * @brief Append a thread node to the list.
180  */
181 static caribou_thread_t* append_thread_node(caribou_thread_t* node)
182 {
183  int state = caribou_interrupts_disable();
184  caribou_thread_t* last = caribou_state.queue;
185  if ( last != NULL )
186  {
187  while ( last->next != NULL )
188  {
189  last = last->next;
190  }
191  last->next = node;
192  }
193  else

196  node->next=NULL;
197  }
198  caribou_interrupts_set(state);
199  return node;

218 static caribou_thread_t* remove_thread_node(caribou_thread_t* node)
219 {
220  int state = caribou_interrupts_disable();

259 /**
260  * @brief Lock the current thread such that it will not yield CPU time until
261  * such time that the thread is unlocked by caribou_thread_unlock(). Needless to say
262  * this should generally be used sparingly.
263  * @return The current state of the lock.
264  */
265 int caribou_thread_lock(void)
266 {
267  int rc=0;
268  int state = caribou_interrupts_disable();
269  if ( caribou_state.current )
270  {
271  rc = caribou_state.current->lock;
272  ++caribou_state.current->lock;
273  }
274  caribou_interrupts_set(state);

278 /**
279  * @brief Unlock a thread which has been previously been locked with caribou_thread_lock()
280  * @return The current state of the lock.
281  */
282 int caribou_thread_unlock(void)
283 {
284  int rc=0;
285  int state = caribou_interrupts_disable();
286  if ( caribou_state.current )
287  {
288  if ( caribou_state.current->lock > 0 )
289  {
290  rc = caribou_state.current->lock;
291  --caribou_state.current->lock;
292  }
293  caribou_interrupts_set(state);
294  if ( !caribou_state.current->lock)
295  caribou_thread_yield();
296  }
297  else
298  {
299  caribou_interrupts_set(state);
300  }
301  return rc;
302 }

320 /**
321  * Sleep another thread for a number of clock ticks.
322  * @param thread The thread to put to sleep.
323  * @param ticks Number of jiffies to sleep for. A zero (0) value indicates forever or until woken up
324  * the caribou_thread_wakeup() function.
325  */
326 void caribou_thread_sleep(caribou_thread_t* thread, caribou_tick_t ticks)
327 {
328  caribou_tick_t start = caribou_timer_ticks();
329  caribou_thread_lock();
330  ++thread->sleep; /* go to sleep, or wakeup if already a wakeup happened */
331  caribou_thread_unlock();
332  while ( thread->sleep > 0) /* while sleeping... */
333  {
334  if ( caribou_timer_ticks_timeout(start,ticks) )
335  {
336  caribou_thread_lock();
337  --thread->sleep; /* wake up the thread */
338  caribou_thread_unlock();
339  break;
340  }
341  caribou_thread_yield();
342  }
343 }

460 /**
461  * Create a new instance of a CARIBOU thread.
462  * @param name The ascii name of the thread. This pointer must remain valid for the life span
463  * of the thread instance.
464  * @param run A Pointer to the entry point function of the thread. Note the entry point function must accept
465  * a void* pointer even if it is not used.
466  * @param finish A function which is called up thread termination.
467  * @param arg An argument pointer which is passed to the thread entry point function run.
468  * @param stackaddr A pointer to bottom of the program stack associated with this thread.
469  * @param stack_size The size of the stack belonging to this thread.
470  * @param priority The priority of the thread.
471  * @return A pointer to the newly created thread or NULL if something failed.
472  */
473 caribou_thread_t* caribou_thread_create(const char* name, void (*run)(void*), void (*finish)(void*), void * arg, void * stackaddr, int stack_size, int16_t priority )
474 {
475  caribou_thread_t* node=NULL;
476  process_frame_t* process_frame;
477 
478  #if defined(CARIBOU_MPU_ENABLED)
479  heap_claim_t claim;
480  int region;
481  int subregion;
482  /* claim a subregion for each thread */
483  while(!heap_mpu_claim(&claim)) {} /* FIXME - maybe we want to fault or something when no more regions? */
484  /* allocate the stack from the subregion */
485  stackaddr = heap_mpu_claim_malloc(&claim, stack_size);
486  #endif
487 
488  node = new_thread_node(caribou_state.current==NULL?caribou_state.queue:caribou_state.current);
489  if ( node != NULL )
490  {
491  #if defined(CARIBOU_MPU_ENABLED)
492  heap_mpu_assign(node,&claim);
493  #endif
494  if ( stackaddr )
495  {
496  uint32_t stack_top;
497  //initialize the process stack pointer
498  //memset(stackaddr,0xFA,stack_size);
499  memset(stackaddr,0x00,stack_size);
500  process_frame = (hw_stack_frame_t *)(stackaddr + stack_size - sizeof(process_frame_t) );
501  memset(process_frame,0,sizeof(process_frame_t));
502  process_frame->hw_stack.r0 = (uint32_t)arg;
503  process_frame->hw_stack.lr = (uint32_t)thread_finish;
504  process_frame->hw_stack.pc = (((uint32_t)run) + INITIAL_PC_OFFSET) & (~1);
505  process_frame->hw_stack.psr = DEFAULT_PSR;
506  #if defined(ARM_FVP_LAZY_STACKING)
507  process_frame->sw_stack.lr = DEFAULT_EXCEPTION_RETURN;
508  #endif
509  stack_top = (stackaddr + stack_size);
510  node->sp = stack_top - sizeof(process_frame_t);
511  node->stack_top = stackaddr + stack_size;
512  node->stack_low = stackaddr;
513  node->stack_low += sizeof(process_frame_t);
514  node->stack_base = stackaddr;
515  }
516  else
517  {
518  node->sp = (&__process_stack_end__);
519  node->stack_top = node->sp;
520  node->stack_low = (&__process_stack_base__);
521  node->stack_low += sizeof(process_frame_t);
522  node->stack_base = (&__process_stack_base__);
523  }
524  node->state = 0;
525  node->name = name;
526  node->arg = arg;
527  node->prio = priority;
528  node->finishfn = finish;
529  append_thread_node(node);
530  }
531  return node;
532 }

617 {

656 static void check_sp(caribou_thread_t* thread)
657 {
658  #if CARIBOU_LOW_STACK_TRAP
659  if ( thread->sp <= thread->stack_low || thread->sp > thread->stack_top )
660  {
661  if ( thread->sp <= thread->stack_low )
662  {
663  _caribou_thread_fault_emit(THREAD_FAULT_STACK_LOW);
664  }
665  else if ( thread->sp <= thread->stack_base )
666  {
667  _caribou_thread_fault_emit(THREAD_FAULT_STACK_OVERFLOW);
668  }
669  else if ( thread->sp > thread->stack_top )
670  {
671  _caribou_thread_fault_emit(THREAD_FAULT_STACK_UNDERFLOW);
672  }
673  }
674  #endif
675  if ( thread->sp < thread->stack_usage || !thread->stack_usage )
676  {
677  thread->stack_usage = thread->sp;
678  }
679 }

681 void caribou_thread_once()
682 {
683  caribou_thread_t* thread=caribou_state.queue;
684  caribou_thread_t* next;
685 
686  // idle hooks...
687  board_idle();
688  // idle the timers...
689  caribou_timer_idle(caribou_state.queue);
690  // terminate threads flagged for termination...
691  for(thread=caribou_state.queue; thread!=NULL; thread=next)
692  {
693  next = thread->next; /* capture the next in case we terminate this thread */
694  check_sp(thread);
695  if ( thread->state & CARIBOU_THREAD_F_TERMINATED && thread != caribou_state.current )
696  {
697  caribou_thread_terminate(thread);
698  }
699  }
700 }

716 /**
717  * @brief Performs the thread scheduling function.
718  * Currently a round-robin search for the next runnable.
719  */
720 static inline void _swap_thread()
721 {
722  /* Preserve the current thread's errno... */
723  caribou_state.current->errno = errno;
724  if ( preemptable( caribou_state.current ) && --caribou_state.priority < 0 )
725  {
726  while( !runnable((caribou_state.current=nextinqueue(caribou_state.current))) )
727  {
728  caribou_state.current->state &= ~CARIBOU_THREAD_F_YIELD;
729  }
730  caribou_state.priority = caribou_state.current->prio;
731  /* Restore the current thread's errno... */
732  errno = caribou_state.current->errno;
733  }
734 }

740  void caribou_thread_mpu_enable(caribou_thread_t* thread)
741  {

772 /**
773  * @brief In the case where the current thread is preempted by caribou_thread_yield(),
774  * then there is no jiffies counting, otherwise it's the same as the normal scheduler
775  * interrupt. If CARIBOU_LOW_STACK_TRAP is defined, stack overflow trapping is performed.
776  */
777 void __attribute__((naked)) _pendsv(void)
778 {
779  pendsv_enter();
780  if ( caribou_state.current )
781  {
782  caribou_state.current->sp = rd_thread_stack_ptr();
783  check_sp(caribou_state.current);
784  caribou_state.current->state |= CARIBOU_THREAD_F_YIELD;
785  /* give up remainder of time slots */
786  caribou_state.priority=-1;
787  #if defined(CARIBOU_MPU_ENABLED)
788  caribou_thread_mpu_enable(caribou_state.current); /* Enable MPU protection (read-only) */
789  #endif
790  _swap_thread();
791  #if defined(CARIBOU_MPU_ENABLED)
792  caribou_thread_mpu_disable(caribou_state.current); /* Disable MPU protection (read-write) */
793  #endif
794  wr_thread_stack_ptr( caribou_state.current->sp );
795  }
796  pendsv_exit();
797 }

799 /**
800  * @brief Entry point for handing a scheduler timer interrupt. Determines
801  * which thread is the next runnable on the queue, and switches context.
802  * The jiffies counter is incremented and a jiffy is added to the current
803  * thread's total run time. If CARIBOU_LOW_STACK_TRAP is defined, stack
804  * overflow trapping is performed.
805  */
806 void __attribute__((naked)) _systick(void)
807 {
808  systick_enter();
809  if ( caribou_state.current )
810  {
811  caribou_state.current->sp = rd_thread_stack_ptr();
812  check_sp(caribou_state.current);
813  ++caribou_state.jiffies;
814  ++caribou_state.current->runtime;
815  #if defined(CARIBOU_MPU_ENABLED)
816  caribou_thread_mpu_enable(caribou_state.current); /* Enable MPU protection (read-only) */
817  #endif
818  _swap_thread();
819  #if defined(CARIBOU_MPU_ENABLED)
820  caribou_thread_mpu_disable(caribou_state.current); /* Disable MPU protection (read-write) */
821  #endif
822  wr_thread_stack_ptr( caribou_state.current->sp );
823  systick_exit();
824  }
825  else
826  {
827  ++caribou_state.jiffies;
828  systick_exit();
829  }
