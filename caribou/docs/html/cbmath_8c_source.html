53 /**
54 * @brief Convert an IEEE 754 number to ASCII
55 * @param buff space for result string
56 * @param conv conversion type 'e', 'f', or 'g' others default to e
57 * @param bsize size of allocated buffer
58 * @param dplace places after the point
59 * @param value value to converted
60 */
61 double dtoa(register char *buff, char conv, int bsize, int dplace, double value)
62 {
63  double modf(), v;
64  register int i;
65  int imax;
66  int j;
67  int exp;
68  int ndigits;
69  int nlead;
70 
71  /* set default value of dplace */
72  if (dplace < 0)
73  dplace = 6;
74 
75  /* strip off sign */
76  if (value < 0.0)
77  {
78  value = -value;
79  *buff++ = '-';
80  }
81 
82  /* scale and compute no of leading digits */
83  if (value == 0.0)
84  imax = PZERO;
85  else
86  {
87  for (imax = PMAX; value < powten[imax] && imax > 0; imax--)
88  {
89  if (conv == 'f' && imax <= PZERO)
90  break;
91  }
92  }
93  exp = imax - PZERO;
94  nlead = exp + 1;
95 
96  /* decide if 'g' goes to 'e' or 'f' */
97  if (conv == 'g')
98  {
99  if (nlead > 6)
100  conv = 'e';
101  else
102  conv = 'f';
103  }
104 
105  /* compute no of digits to print */
106  /* change 'f' to 'e' if insufficient space for result */
107  if (conv == 'f')
108  {
109  ndigits = dplace + nlead;
110  if (ndigits + 3 > bsize)
111  {
112  conv = 'e';
113  dplace = SIGFIGS;
114  }
115  }
116  if (conv != 'f')
117  {
118  nlead = 1;
119  ndigits = dplace + nlead;
120  if (ndigits > SIGFIGS)
121  ndigits = SIGFIGS;
122  }
123 
124  /* scale to range 1.0 - 10.0 and round up */
125  if (conv == 'e' && imax == 0 && value < powten[0])
126  {
127  value *= 10.0;
128  exp--;
129  }
130  value = value / powten[imax] + 0.5 * powten[PZERO - ndigits + 1];
131  value = modf(value, &v);
132  if (v >= 10.0)
133  {
134  *buff++ = '1';
135  v -= 10.0;
136  }
137 
138  /* build digit string */
139  for (i = 0; i < ndigits; i++)
140  {
141  if (i == nlead)
142  *buff++ = '.';
143  *buff++ = (int)v + '0';
144  value = modf(value * 10.0, &v);
145  //thread_yield();
146  }
147 
148  /* if not 'f' print exponent part */
149  if (conv != 'f')
150  {
151  *buff++ = 'e';
152  if (exp < 0)
153  {
154  exp = -exp;
155  *buff++ = '-';
156  }
157  else
158  *buff++ = '+';
159  j = exp/10;
160  *buff++ = '0' + j;
161  *buff++ = '0' + exp - 10*j;
162  }
163  /* print final null */
164  *buff++ = '\0';
165 
166  return value;
167 }

169 /**
170 * @brief convert a string to double
171 * @return resulting double
172 */
173 double atof(const char* s)
174 {
175  double val;
176  int i, j, sign, esign, ex, ex1;
177 
178  /* skip white space */
179  for (i=0; s[i]==' ' || s[i]=='\n' || s[i]=='\t'; i++);
180  /* set sign to be +1 or -1 */
181  sign = 1;
182  if (s[i] == '+' || s[i] == '-') /* sign */
183  sign = (s[i++]=='+') ? 1 : -1;
184  /* read digits and build value for as long as there are digits */
185  ex = PZERO - 1;
186  for (val = 0.0, j = PZERO; s[i] >= '0' && s[i] <= '9'; i++)
187  {
188  if (j >= 0)
189  val += powten[j] * (s[i] - '0');
190  j--;
191  ex++;
192  }
193  /* if first non-digit is decimal point skip it and keep going */
194  /* if it is not a decimal point we fall through to test for exponent */
195  if (s[i] == '.')
196  i++;
197  /* continue to build value while the digits keep coming */
198  for (; s[i] >= '0' && s[i] <= '9'; i++)
199  {
200  if (j >= 0)
201  val += powten[j] * (s[i] - '0');
202  j--;
203  }
204  /* if non-digit was an exponent flag deal with exponent */
205  if (s[i]=='e' || s[i]=='E' || s[i]=='d' || s[i]=='D')
206  {
207  i++;
208  esign = 1;
209  /* check for explicit + or - sign in exponent and deal with it */
210  if (s[i] == '+' || s[i] == '-')
211  esign = (s[i++]=='+') ? 1 : -1;
212  for (ex1 = 0; s[i] >= '0' && s[i] <= '9'; i++)
213  ex1 = 10 * ex1 + s[i] - '0';
214  ex += ex1 * esign;
215  }
216  /* incorporate exponent into val */
217  if (ex < 0)
218  val = 0.0;
219  else if (ex < PMAX || (ex == PMAX && val < HUGE_NOR))
220  val *= powten[ex];
221  else
222  {
223  //cmemsg(FP_BIGI, s);
224  val = HUGE;
225  }
226  /* check that we have an acceptable termination; if not call error system
227  before returning with what we have */
228  if (s[i] != ' ' && s[i] != '\t' && s[i] != '\n' && s[i] != '\0')
229  {
230  /* cmemsg(FP_BADC, s) */
231  }
232  return sign*val;
233 }

237 /* log10(x)
238  * Return the base 10 logarithm of x
239  *
240  * Method :
241  * Let log10_2hi = leading 40 bits of log10(2) and
242  * log10_2lo = log10(2) - log10_2hi,
243  * ivln10 = 1/log(10) rounded.
244  * Then
245  * n = ilogb(x),
246  * if(n<0) n = n+1;
247  * x = scalbn(x,-n);
248  * log10(x) := n*log10_2hi + (n*log10_2lo + ivln10*log(x))
249  *
250  * Note 1:
251  * To guarantee log10(10**n)=n, where 10**n is normal, the rounding
252  * mode must set to Round-to-Nearest.
253  * Note 2:
254  * [1/log(10)] rounded to 53 bits has error .198 ulps;
255  * log10 is monotonic at all binary break points.
256  *
257  * Special cases:
258  * log10(x) is NaN with signal if x < 0;
259  * log10(+INF) is +INF with no signal; log10(0) is -INF with signal;
260  * log10(NaN) is that NaN with no signal;
261  * log10(10**N) = N for N=0,1,...,22.
262  *
263  * Constants:
264  * The hexadecimal values are the intended ones for the following constants.
265  * The decimal values may be used, provided that the compiler will convert
266  * from decimal to binary accurately enough to produce the hexadecimal values
267  * shown.
268  */
269 double log10(double x)
270 {
271  double y,z;
272  int32_t i,k,hx;
273  uint32_t lx;
274 
275  EXTRACT_WORDS(hx,lx,x);
276 
277  k=0;
278  if (hx < 0x00100000) /* x < 2**-1022 */
279  {
280  if (((hx&0x7fffffff)|lx)==0)
281  return -two54/zero; /* log(+-0)=-inf */
282  if (hx<0) return (x-x)/zero; /* log(-#) = NaN */
283  k -= 54; x *= two54; /* subnormal number, scale up x */
284  GET_HIGH_WORD(hx,x);
285  }
286  if (hx >= 0x7ff00000) return x+x;
287  k += (hx>>20)-1023;
288  i = ((uint32_t)k&0x80000000)>>31;
289  hx = (hx&0x000fffff)|((0x3ff-i)<<20);
290  y = (double)(k+i);
291  SET_HIGH_WORD(x,hx);
292  z = y*log10_2lo + ivln10*log(x);
293  return z+y*log10_2hi;
294 }

297 /* log(x)
298  * Return the logrithm of x
299  *
300  * Method :
301  * 1. Argument Reduction: find k and f such that
302  * x = 2^k * (1+f),
303  * where sqrt(2)/2 < 1+f < sqrt(2) .
304  *
305  * 2. Approximation of log(1+f).
306  * Let s = f/(2+f) ; based on log(1+f) = log(1+s) - log(1-s)
307  * = 2s + 2/3 s**3 + 2/5 s**5 + .....,
308  * = 2s + s*R
309  * We use a special Reme algorithm on [0,0.1716] to generate
310  * a polynomial of degree 14 to approximate R The maximum error
311  * of this polynomial approximation is bounded by 2**-58.45. In
312  * other words,
313  * 2 4 6 8 10 12 14
314  * R(z) ~ Lg1*s +Lg2*s +Lg3*s +Lg4*s +Lg5*s +Lg6*s +Lg7*s
315  * (the values of Lg1 to Lg7 are listed in the program)
316  * and
317  * | 2 14 | -58.45
318  * | Lg1*s +...+Lg7*s - R(z) | <= 2
319  * | |
320  * Note that 2s = f - s*f = f - hfsq + s*hfsq, where hfsq = f*f/2.
321  * In order to guarantee error in log below 1ulp, we compute log
322  * by
323  * log(1+f) = f - s*(f - R) (if f is not too large)
324  * log(1+f) = f - (hfsq - s*(hfsq+R)). (better accuracy)
325  *
326  * 3. Finally, log(x) = k*ln2 + log(1+f).
327  * = k*ln2_hi+(f-(hfsq-(s*(hfsq+R)+k*ln2_lo)))
328  * Here ln2 is split into two floating point number:
329  * ln2_hi + ln2_lo,
330  * where n*ln2_hi is always exact for |n| < 2000.
331  *
332  * Special cases:
333  * log(x) is NaN with signal if x < 0 (including -INF) ;
334  * log(+INF) is +INF; log(0) is -INF with signal;
335  * log(NaN) is that NaN with no signal.
336  *
337  * Accuracy:
338  * according to an error analysis, the error is always less than
339  * 1 ulp (unit in the last place).
340  *
341  * Constants:
342  * The hexadecimal values are the intended ones for the following
343  * constants. The decimal values may be used, provided that the
344  * compiler will convert from decimal to binary accurately enough
345  * to produce the hexadecimal values shown.
346  */

350  int32_t k,hx,i,j;

352

360  if (hx<0) return (x-x)/zero; /* log(-#) = NaN */
361  k -= 54; x *= two54; /* subnormal number, scale up x */
362  GET_HIGH_WORD(hx,x);
363  }
364  if (hx >= 0x7ff00000)
365  return x+x;
366  k += (hx>>20)-1023;
367  hx &= 0x000fffff;
368  i = (hx+0x95f64)&0x100000;
369  SET_HIGH_WORD(x,hx|(i^0x3ff00000)); /* normalize x or x/2 */
370  k += (i>>20);

373  {

378  else

382  }
383  }

385  if(k==0)

387  else
388  {
389  dk=(double)k;
390  return dk*ln2_hi-((R-dk*ln2_lo)-f);
391  }
392  }
393  s = f/(2.0+f);

395  z = s*s;

397  w = z*z;
398  j = 0x6b851-hx;

400  t2= z*(Lg1+w*(Lg3+w*(Lg5+w*Lg7)));
401  i |= j;
402  R = t2+t1;
403  if(i>0)
404  {
405  hfsq=0.5*f*f;
406  if(k==0)
407  return f-(hfsq-s*(hfsq+R));
408  else
409  return dk*ln2_hi-((hfsq-(s*(hfsq+R)+dk*ln2_lo))-f);
410  }
411  else
412  {
413  if(k==0)
414  return f-s*(f-R);
415  else
416  return dk*ln2_hi-((s*(f-R)-dk*ln2_lo)-f);
417  }
418 }
