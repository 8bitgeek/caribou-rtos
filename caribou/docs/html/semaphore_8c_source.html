66 /** **************************************************************************

71 bool caribou_semaphore_wait(caribou_semaphore_t* semaphore, caribou_tick_t timeout)
72 {
73  bool rc=true;
74  caribou_tick_t start = caribou_timer_ticks();
75  while ( !(rc=caribou_semaphore_try_wait(semaphore)) )

87 /** **************************************************************************
88  ** @brief This operation decreases the semaphore counter
89  ** @param semaphore The semaphore to operate on.
90  ** @return true if the semaphore was released.
91  *****************************************************************************/
92 bool caribou_semaphore_try_wait(caribou_semaphore_t* semaphore)
93 {
94  bool rc=false;
95  int state = caribou_interrupts_disable();
96  if ( semaphore->count > 0 )
97  {
98  --semaphore->count; /* release */
99  rc=true;
100  }
101  caribou_interrupts_set(state);
102  return rc;
