72 /**
73  * Try to acquire lock. If the mutex is currently held, the calling thread
74  * will block unless the current thread is the holder of the current lock.
75  * If the current already holds the lock,
76  * then a new lock will be granted if the CARIBOU_MUTEX_F_RECURSIVE flasg is set.
77  * @param A pointer to in instance of a caribou_mutex_t structure.
78  * @param timeout The timeout to wait in milliseconds, or forever if zero.
79  * @return true if the lock was acquired.
80  * @note If the current thread currently holds a lock on the mutex, then a new lock will be granted, however, in order
81  * to release the lock completely such that it may be acquired by another thread, then caribou_mutex_unlock() must be called a number of time equal to the number
82  * of times a lock was acquired by the same (current) thread.
83  */
84 bool caribou_mutex_lock(caribou_mutex_t* mutex,uint32_t timeout)
85 {
86  caribou_timer_t timer;
87  caribou_thread_lock(); /* lock the thread if it's not locked already */
88  ++mutex->blocking; /* tell the holder of the lock that this thread is blocking */
89  caribou_thread_unlock(); /* restore previous scheduler lock state */
90  if ( timeout )
91  {
92  caribou_timer_set(&timer,from_ms(timeout));
93  caribou_timer_init(&timer,NULL,NULL,CARIBOU_TIMER_F_ONESHOT);
94  }
95  while ( !caribou_mutex_trylock(mutex) && (!timeout || (timeout && !caribou_timer_expired(&timer))))
96  {
97  if ( (mutex->flags & CARIBOU_MUTEX_F_NOYIELD) == 0 )
98  {
99  caribou_thread_yield();
100  }
101  }
102  caribou_thread_lock(); /* lock the scheduler if it's not locked already */
103  --mutex->blocking; /* tell the holder of the lock that this thread is unblocking */
104  caribou_thread_unlock(); /* restore the previous scheduler state */
105  return true;
106 }

108 /**
109  * Try to acquire a mutex lock without blocking. If the mutex is currently held, the calling thread
110  * will not block, however, caribou_mutex_trylock() will return false, and the lock will not have been
111  * acquired by the calling thread.
112  * @param A pointer to in instance of a caribou_mutex_t structure.
113  * @return true if the lock was acquired.
114  * @note If the current thread currently holds a lock on the mutex, then a new lock will be granted, however, in order
115  * to release the lock completely such that it may be acquired by another thread, then caribou_mutex_unlock() must be called a number of time equal to the number
116  * of times a lock was acquired by the same (current) thread.
117  */
118 bool caribou_mutex_trylock(caribou_mutex_t* mutex)
119 {
120  bool rc;
121  caribou_thread_t* current_thread = caribou_thread_current();
122  caribou_thread_lock();
123  if ( mutex->locks )
124  {
125  if ( mutex->thread != current_thread )
126  {
127  rc = false; /* qcquired */
128  }
129  else if ( mutex->flags & CARIBOU_MUTEX_F_RECURSIVE )
130  {
131  ++mutex->locks; // increment number of locks
132  rc = true; /* acquired */
133  }
134  else
135  {
136  rc = false; /* not acquired */
137  }
138  }
139  else
140  {
141  /* acquired */
142  ++mutex->locks; // increment number of locks
143  mutex->thread = current_thread;
144  rc = true;
145  }
146  caribou_thread_unlock();
147  return rc;
148 }

150 /**
151  * @brief Unlock a mutex instance which was previously acquired with thread_mutex_lock() or thread_caribou_mutex_trylock()
152  * If there are other threads that are currently blocking on the mutex, then caribou_mutex_unlock() will yield the thread after
153  * releasing the mutex, otherwise, if there are no other threads blocking on the mutex, then continue without yielding.
154  * @param mutex A mutex instance which was previously locked.
155  * @return Boolean, true if the lock was released, otherwise false.
156  * @note A mutex can only be unlocked by the thread which locked it.
157  */
158 bool caribou_mutex_unlock(caribou_mutex_t* mutex)
159 {
160  bool rc=false;
161  caribou_thread_lock();
162  if ( mutex->locks && mutex->thread == caribou_thread_current() )
163  {
164  if ( --mutex->locks == 0 ) // decrement number of locks.
165  mutex->thread=NULL;
166  rc=true;
167  }
168  // if there are blockers, then yield, otherwise not...
169  if ( mutex->blocking )
170  {
171  caribou_thread_unlock();
172  if ( (mutex->flags & CARIBOU_MUTEX_F_NOYIELD) == 0 )
173  {
174  caribou_thread_yield();
175  }
176  }
177  else
178  {
179  caribou_thread_unlock();
180  }
181  return rc;
182 }
