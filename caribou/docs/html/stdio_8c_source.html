139 {

173 static int printi(FILE *fp, char **out, long i, int b, int sg, int width, int pad, int letbase)
174 {
175  char print_buf[PRINT_BUF_LEN];
176  char *s;
177  long t, neg = 0, pc = 0;
178  unsigned long u = i;
179 
180  if (i == 0)
181  {
182  print_buf[0] = '0';
183  print_buf[1] = '\0';
184  return prints (fp, out, print_buf, width, pad);
185  }
186 
187  if (sg && b == 10 && i < 0)
188  {
189  neg = 1;
190  u = -i;
191  }
192 
193  s = print_buf + PRINT_BUF_LEN-1;
194  *s = '\0';
195 
196  while (u)
197  {
198  t = u % b;
199  if( t >= 10 )
200  t += letbase - '0' - 10;
201  *--s = t + '0';
202  u /= b;
203  }
204 
205  if (neg)
206  {
207  if( width && (pad & PAD_ZERO) ) {
208  printchar (fp, out, '-');
209  ++pc;
210  --width;
211  }
212  else {
213  *--s = '-';
214  }
215  }
216 
217  return pc + prints (fp, out, s, width, pad);
218 }

220 static int print(FILE *fp, char **out, const char *format, va_list args )
221 {
222  int width, pad;
223  int pc = 0;
224  char scr[2];
225 
226  for (; *format != 0; ++format)
227  {
228  if (*format == '%')
229  {
230  ++format;
231  width = pad = 0;
232  if (*format == '\0') break;
233  if (*format == '%') goto out;
234  if (*format == '-') {
235  ++format;
236  pad = PAD_RIGHT;
237  }
238  while (*format == '0')
239  {
240  ++format;
241  pad |= PAD_ZERO;
242  }
243  for ( ; *format >= '0' && *format <= '9'; ++format)
244  {
245  width *= 10;
246  width += *format - '0';
247  }
248  if( *format == 's' )
249  {
250  register char *s = (char *)va_arg( args, int );
251  pc += prints(fp, out, s?s:"(null)", width, pad);
252  continue;
253  }
254  if( *format == 'd' )
255  {
256  pc += printi(fp, out, va_arg( args, int ), 10, 1, width, pad, 'a');
257  continue;
258  }
259  if( *format == 'l' )
260  {
261  pc += printi(fp, out, va_arg( args, long ), 10, 1, width, pad, 'a');
262  continue;
263  }
264  if( *format == 'x' )
265  {
266  pc += printi(fp, out, va_arg( args, int ), 16, 0, width, pad, 'a');
267  continue;
268  }
269  if( *format == 'X' || *format == 'p' )
270  {
271  pc += printi(fp, out, va_arg( args, int ), 16, 0, width, pad, 'A');
272  continue;
273  }
274  if( *format == 'u' )
275  {
276  pc += printi(fp, out, va_arg( args, int ), 10, 0, width, pad, 'a');
277  continue;
278  }
279  if( *format == 'c' )
280  {
281  /* char are converted to int then pushed on the stack */
282  scr[0] = (char)va_arg( args, int );
283  scr[1] = '\0';
284  pc += prints (fp, out, scr, width, pad );
285  continue;
286  }
287  #if defined(CARIBOU_PRINTF_DOUBLE)
288  if( *format == 'g' || *format == 'f' )
289  {
290  char temp[48];
291  dtoa(temp,'f',48,9,(double)va_arg( args, double ));
292  pc += prints(fp, out, temp, width, pad);
293  continue;
294  }
295  #endif
296  }
297  else
298  {
299  out:
300  printchar(fp, out, *format);
301  ++pc;
302  }
303  }
304  if (out)
305  {
306  **out = '\0';
307  }
308  return pc;
309 }
