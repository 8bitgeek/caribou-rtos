66 /*
67  * _atob(vp,p,base)
68  */
69 static int _atob (unsigned long long *vp, char *p, int base)
70 {
71  unsigned long long value, v1, v2;
72  char *q, tmp[20];
73  int digit;
74 
75  if (p[0] == '0' && (p[1] == 'x' || p[1] == 'X')) {
76  base = 16;
77  p += 2;
78  }
79 
80  if (base == 16 && (q = strchr (p, '.')) != 0) {
81  if (q - p > sizeof(tmp) - 1)
82  return (0);
83 
84  strncpy (tmp, p, q - p);
85  tmp[q - p] = '\0';
86  if (!_atob (&v1, tmp, 16))
87  return (0);
88 
89  q++;
90  if (strchr (q, '.'))
91  return (0);
92 
93  if (!_atob (&v2, q, 16))
94  return (0);
95  *vp = (v1 << 16) + v2;
96  return (1);
97  }
98 
99  value = *vp = 0;
100  for (; *p; p++) {
101  if (*p >= '0' && *p <= '9')
102  digit = *p - '0';
103  else if (*p >= 'a' && *p <= 'f')
104  digit = *p - 'a' + 10;
105  else if (*p >= 'A' && *p <= 'F')
106  digit = *p - 'A' + 10;
107  else
108  return (0);
109 
110  if (digit >= base)
111  return (0);
112  value *= base;
113  value += digit;
114  }
115  *vp = value;
116  return (1);
117 }

150  * char *btoa(dst,value,base)
151  * converts value to ascii, result in dst
152  */
153 char *btoa(char *dst, uint32_t value, int base)
154 {
155  char buf[34], digit;
156  int i, j, rem, neg;
157 
158  if (value == 0) {
159  dst[0] = '0';
160  dst[1] = 0;
161  return (dst);
162  }
163 
164  neg = 0;
165  if (base == -10) {
166  base = 10;
167  if (value & (1L << 31)) {
168  value = (~value) + 1;
169  neg = 1;
170  }
171  }
172 
173  for (i = 0; value != 0; i++) {
174  rem = value % base;
175  value /= base;
176  if (rem >= 0 && rem <= 9)
177  digit = rem + '0';
178  else if (rem >= 10 && rem <= 36)
179  digit = (rem - 10) + 'a';
180  buf[i] = digit;
181  }
182 
183  buf[i] = 0;
184  if (neg)
185  strcat (buf, "-");
186 
187  /* reverse the string */
188  for (i = 0, j = strlen (buf) - 1; j >= 0; i++, j--)
189  dst[i] = buf[j];
190  dst[i] = 0;
191  return (dst);
192 }

240  * gethex(vp,p,n)
241  * convert n hex digits from p to binary, result in vp,
242  * rtn 1 on success
243  */
244 int gethex(int32_t *vp, char *p, int n)
245 {
246  unsigned long v;
247  int digit;
248 
249  for (v = 0; n > 0; n--) {
250  if (*p == 0)
251  return (0);
252  if (*p >= '0' && *p <= '9')
253  digit = *p - '0';
254  else if (*p >= 'a' && *p <= 'f')

256  else if (*p >= 'A' && *p <= 'F')
257  digit = *p - 'A' + 10;

262  v |= digit;

264  }

419 {

453 static int printi(FILE *fp, char **out, long i, int b, int sg, int width, int pad, int letbase)
454 {
455  char print_buf[PRINT_BUF_LEN];
456  char *s;
457  long t, neg = 0, pc = 0;
458  unsigned long u = i;
459 
460  if (i == 0)
461  {
462  print_buf[0] = '0';
463  print_buf[1] = '\0';
464  return prints (fp, out, print_buf, width, pad);
465  }
466 
467  if (sg && b == 10 && i < 0)
468  {
469  neg = 1;
470  u = -i;
471  }
472 
473  s = print_buf + PRINT_BUF_LEN-1;
474  *s = '\0';
475 
476  while (u)
477  {
478  t = u % b;
479  if( t >= 10 )
480  t += letbase - '0' - 10;
481  *--s = t + '0';
482  u /= b;
483  }
484 
485  if (neg)
486  {
487  if( width && (pad & PAD_ZERO) ) {
488  printchar (fp, out, '-');
489  ++pc;
490  --width;
491  }
492  else {
493  *--s = '-';
494  }
495  }
496 
497  return pc + prints (fp, out, s, width, pad);
498 }

500 static int print(FILE *fp, char **out, const char *format, va_list args )
501 {
502  int width, pad;
503  int pc = 0;
504  char scr[2];
505 
506  for (; *format != 0; ++format)
507  {
508  if (*format == '%')
509  {
510  ++format;
511  width = pad = 0;
512  if (*format == '\0') break;
513  if (*format == '%') goto out;
514  if (*format == '-') {
515  ++format;
516  pad = PAD_RIGHT;
517  }
518  while (*format == '0')
519  {
520  ++format;
521  pad |= PAD_ZERO;
522  }
523  for ( ; *format >= '0' && *format <= '9'; ++format)
524  {
525  width *= 10;
526  width += *format - '0';
527  }
528  if( *format == 's' )
529  {
530  char *s = (char *)va_arg( args, char* );
531  pc += prints(fp, out, s?s:"(null)", width, pad);
532  continue;
533  }
534  if( *format == 'd' )
535  {
536  pc += printi(fp, out, va_arg( args, int ), 10, 1, width, pad, 'a');
537  continue;
538  }
539  if( *format == 'l' )
540  {
541  pc += printi(fp, out, va_arg( args, long ), 10, 1, width, pad, 'a');
542  continue;
543  }
544  if( *format == 'x' )
545  {
546  pc += printi(fp, out, va_arg( args, int ), 16, 0, width, pad, 'a');
547  continue;
548  }
549  if( *format == 'X' || *format == 'p' )
550  {
551  pc += printi(fp, out, va_arg( args, int ), 16, 0, width, pad, 'A');
552  continue;
553  }
554  if( *format == 'u' )
555  {
556  pc += printi(fp, out, va_arg( args, int ), 10, 0, width, pad, 'a');
557  continue;
558  }
559  if( *format == 'c' )
560  {
561  /* char are converted to int then pushed on the stack */
562  scr[0] = (char)va_arg( args, int );
563  scr[1] = '\0';
564  pc += prints (fp, out, scr, width, pad );
565  continue;
566  }
567  #if defined(CARIBOU_PRINTF_DOUBLE)
568  if( *format == 'g' )
569  {
570  char temp[48];
571  dtoa(temp,'f',48,3,(double)va_arg( args, double ));
572  pc += prints(fp, out, temp, width, pad);
573  continue;
574  }
575  if( *format == 'f' )
576  {
577  char temp[48];
578  float f = (float)va_arg( args, float );
579  double d = f;
580  dtoa(temp,'f',48,3,d);
581  pc += prints(fp, out, temp, width, pad);
582  continue;
583  }
584  #endif
585  }
586  else
587  {
588  out:
589  printchar(fp, out, *format);
590  ++pc;
591  }
592  }
593  if (out)
594  {
595  **out = '\0';
596  }
597  return pc;
598 }

656 /*
657  * vsscanf(buf,fmt,ap)
658  */
659 int vsscanf(const char *buf, const char *s, va_list ap)
660 {
661  int count, noassign, width, base, lflag;
662  const char *tc;
663  char *t, tmp[MAXLN];
664 
665  base=10;
666  count = noassign = width = lflag = 0;
667  while (*s && *buf) {
668  while (isspace (*s))
669  s++;
670  if (*s == '%') {
671  s++;
672  for (; *s; s++) {
673  if (strchr ("dibouxcsefg%", *s))
674  break;
675  if (*s == '*')
676  noassign = 1;
677  else if (*s == 'l' || *s == 'L')
678  lflag = 1;
679  else if (*s >= '1' && *s <= '9') {
680  for (tc = s; isdigit (*s); s++);
681  strncpy (tmp, tc, s - tc);
682  tmp[s - tc] = '\0';
683  atob (&width, tmp, 10);
684  s--;
685  }
686  }
687  if (*s == 's') {
688  while (isspace (*buf))
689  buf++;
690  if (!width)
691  width = strcspn (buf, ISSPACE);
692  if (!noassign) {
693  strncpy (t = va_arg (ap, char *), buf, width);
694  t[width] = '\0';
695  }
696  buf += width;
697  } else if (*s == 'c') {
698  if (!width)
699  width = 1;
700  if (!noassign) {
701  strncpy (t = va_arg (ap, char *), buf, width);
702  t[width] = '\0';
703  }
704  buf += width;
705  } else if (strchr ("dobxu", *s)) {
706  while (isspace (*buf))
707  buf++;
708  if (*s == 'd' || *s == 'u')
709  base = 10;
710  else if (*s == 'x')
711  base = 16;
712  else if (*s == 'o')
713  base = 8;
714  else if (*s == 'b')
715  base = 2;
716  if (!width) {
717  if (isspace (*(s + 1)) || *(s + 1) == 0)
718  width = strcspn (buf, ISSPACE);
719  else
720  width = strchr (buf, *(s + 1)) - buf;
721  }
722  strncpy (tmp, buf, width);
723  tmp[width] = '\0';
724  buf += width;
725  if (!noassign)
726  atob (va_arg (ap, uint32_t *), tmp, base);
727  }
728  if (!noassign)
729  count++;
730  width = noassign = lflag = 0;
731  s++;
732  } else {
733  while (isspace (*buf))
734  buf++;
735  if (*s != *buf)
736  break;
737  else
738  s++, buf++;
739  }
740  }
741  return (count);
742 }
