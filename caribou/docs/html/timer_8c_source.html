53 // append a thread node to the list
54 static caribou_timer_t* append_timer_node(caribou_thread_t* thread, caribou_timer_t* node)
55 {
56  int state = caribou_lock();
57  register caribou_timer_t* last = thread->timer;
58  if ( last != NULL )
59  {
60  while ( last->next != NULL )
61  {
62  last = last->next;
63  }
64  last->next = node;
65  node->prev = last;
66  }
67  else
68  {
69  thread->timer = node;
70  node->prev=NULL;
71  node->next=NULL;
72  }
73  caribou_lock_set(state);
74  return node;
75 }

77 // remove a timer node from the list
78 static caribou_timer_t* remove_timer_node(caribou_thread_t* thread, caribou_timer_t* node)
79 {
80  int state = caribou_lock();
81  if ( node == thread->timer )
82  {
83  thread->timer = node->next;
84  if ( thread->timer != NULL )
85  {
86  thread->timer->prev = NULL;
87  }
88  }
89  else
90  {
91  if ( node->prev != NULL ) node->prev->next = node->next;
92  if ( node->next != NULL ) node->next->prev = node->prev;
93  node->next=NULL;
94  node->prev=NULL;
95  }
96  caribou_lock_set(state);
97  return node;
98 }

162 /**
163  * @brief Determine of the timeout period has expired in relationn to the start time.
164  * @param start The time from which the timeout period should have started.
165  * @param timeout The timeout period, TIMEOUT_INFINATE wait forever, TIMEOUT_IMMEDIATE, wait never.
166  * @return true if the timeout has expired.
167  */
168 bool caribou_timer_ticks_timeout(caribou_tick_t start, caribou_tick_t timeout)
169 {
170  bool rc=false;
171  if ( TIMEOUT_INFINITE != timeout )
172  {
173  if ( TIMEOUT_IMMEDIATE != timeout )
174  {
175  rc = ((caribou_timer_ticks()-start) >= timeout);
176  }
177  else
178  {
179  rc=true;
180  }
181  }
182  return rc;
183 }

233 /**
234 ** @brief Expire sleep timers and timer timers
235 ** @param thread The subject of the timer and sleep expiry.
236 ** @param delta_jiffies The number of jiffies elapsed since the last timer expiry run.
237 ** @return number timers expired.
238 */
239 static int expire_timer(caribou_thread_t* thread, caribou_timer_t* timer, int delta_jiffies)
240 {
241  int rc=0;
242  if ( timer->ticks )
243  {
244  timer->ticks = (timer->ticks>=delta_jiffies)?timer->ticks-delta_jiffies:0;
245  if ( !timer->ticks )
246  {
247  if ( timer->timerfn )
248  {
249  timer->timerfn(thread,timer,timer->fnarg);
250  }
251  if ( !(timer->flags & CARIBOU_TIMER_F_ONESHOT) )
252  {
253  timer->ticks = timer->reloadticks;
254  }
255  ++rc;
256  }
257  }
258  return rc;
259 }

261 /// main thread idle processing
262 /// @return number of timers expired
263 int caribou_timer_idle(caribou_thread_t* thread)
264 {
265  int rc;
266  int delta_jiffies;
267  int state = chip_systick_irq_disable();
268  delta_jiffies=(_caribou_jiffies-_caribou_tail_jiffies);
269  if ( delta_jiffies )
270  {
271  caribou_thread_t* next;
272  caribou_timer_t* timer;
273  _caribou_tail_jiffies = _caribou_jiffies;
274  for(; thread!=NULL; thread=next)
275  {
276  /** expire timers...*/
277  for(timer=((caribou_thread_t*)thread)->timer; timer!=NULL; timer = timer->next)
278  {
279  rc += expire_timer(thread,timer,delta_jiffies);
280  }
281  next = thread->next; /* capture the next in case we terminate this thread */
282  }
283  }
284  chip_systick_irq_set(state);
285  return rc;
286 }
