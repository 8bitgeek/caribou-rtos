58 static bool find_timer_node(caribou_thread_t* thread, caribou_timer_t* node)
59 {
60  bool rc=false;
61  caribou_mutex_lock(&timer_mutex,0);
62  for( caribou_timer_t* next=thread->timer; !rc && next!=NULL; next=next->next)
63  {
64  if ( node == next )
65  {
66  rc=true;
67  }
68  }
69  caribou_mutex_unlock(&timer_mutex);
70  return rc;
71 }

73 // append a thread node to the list
74 static caribou_timer_t* append_timer_node(caribou_thread_t* thread, caribou_timer_t* node)
75 {
76  if ( !find_timer_node(thread,node) )
77  {
78  caribou_mutex_lock(&timer_mutex,0);
79  caribou_timer_t* last = thread->timer;
80  if ( last != NULL )
81  {
82  while ( last->next != NULL )
83  {
84  last = last->next;
85  }
86  last->next = node;
87  node->prev = last;
88  }
89  else
90  {
91  thread->timer = node;
92  node->prev=NULL;
93  node->next=NULL;
94  }
95  caribou_mutex_unlock(&timer_mutex);
96  }
97  return node;
98 }

100 // remove a timer node from the list
101 static caribou_timer_t* remove_timer_node(caribou_thread_t* thread, caribou_timer_t* node)
102 {
103  caribou_mutex_lock(&timer_mutex,0);
104  if ( node == thread->timer )
105  {
106  thread->timer = node->next;
107  if ( thread->timer != NULL )
108  {
109  thread->timer->prev = NULL;
110  }
111  }
112  else
113  {
114  if ( node->prev != NULL ) node->prev->next = node->next;
115  if ( node->next != NULL ) node->next->prev = node->prev;
116  node->next=NULL;
117  node->prev=NULL;
118  }
119  caribou_mutex_unlock(&timer_mutex);
120  return node;
121 }

188 /**
189  * @brief Determine of the timeout period has expired in relationn to the start time.
190  * @param start The time from which the timeout period should have started.
191  * @param timeout The timeout period, TIMEOUT_INFINATE wait forever, TIMEOUT_IMMEDIATE, wait never.
192  * @return true if the timeout has expired.
193  */
194 bool caribou_timer_ticks_timeout(caribou_tick_t start, caribou_tick_t timeout)
195 {
196  if ( TIMEOUT_INFINITE == timeout ) /* never timed out */
197  return false;
198  if ( TIMEOUT_IMMEDIATE == timeout ) /* always timed out */
199  return true;
200  return ((caribou_state.jiffies-start) >= timeout);
201 }

254 /**
255 ** @brief Expire sleep timers and timer timers
256 ** @param thread The subject of the timer and sleep expiry.
257 ** @param delta_jiffies The number of jiffies elapsed since the last timer expiry run.
258 ** @return number timers expired.
259 */
260 static int expire_timer(caribou_thread_t* thread, caribou_timer_t* timer, int delta_jiffies)
261 {
262  int rc=0;
263  if ( timer->ticks )
264  {
265  timer->ticks = (timer->ticks>=delta_jiffies)?timer->ticks-delta_jiffies:0;
266  if ( !timer->ticks )
267  {
268  if ( timer->timerfn )
269  {
270  timer->timerfn(thread,timer,timer->fnarg);
271  }
272  if ( !(timer->flags & CARIBOU_TIMER_F_ONESHOT) )
273  {
274  timer->ticks = timer->reloadticks;
275  }
276  ++rc;
277  }
278  }
279  return rc;
280 }

282 /**
283  * @brief Used by CARIBOU to process timers during the main thread idle loop.
284  * @return number of timers expired
285  */
286 int caribou_timer_idle(caribou_thread_t* thread)
287 {
288  int rc;
289  int delta_jiffies;
290  caribou_mutex_lock(&timer_mutex,0);
291  delta_jiffies=(caribou_state.jiffies-caribou_state.tail_jiffies);
292  if ( delta_jiffies )
293  {
294  caribou_state.tail_jiffies = caribou_state.jiffies;
295  while(thread)
296  {
297  /** expire timers...*/
298  for(caribou_timer_t* timer=thread->timer; timer!=NULL; timer = timer->next)
299  {
300  rc += expire_timer(thread,timer,delta_jiffies);
301  }
302  thread = thread->next;
303  }
304  }
305  caribou_mutex_unlock(&timer_mutex);
306  return rc;
307 }
