27 /**
28  * @brief Call into the kernel from process thread space.
29  * @param fn Function code, indicates the ind of operation.
30  * @param arg The argument pointer to data related to the kind of operation.
31  * @return 0 upon success. -1 on errno.
32  */
33 int kcall(kcall_t fn, void* arg)
34 {
35  int rc;
36  __asm__ __volatile__( " mov r4,%1 \n" /* R4=fn */
37  " mov r5,%2 \n" /* R5=arg */
38  " svc #0 \n" /* Software Interrupt*/
39  " mov %0,r4 \n" /* rc=R4 */
40  : "=r" (rc)
41  : "r" (fn), "r" (arg)
42  : "r4","r5"
43  );
44  return rc;
45 }

116 {
