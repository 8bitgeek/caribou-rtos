49 /**
50  * @brief Determine if a vector is already installed.
51  * @param vector The device specific interrupt vector number.
52  * @param isr A pointer to the interrupt service routing for the vector.
53  * @param arg An optional argument to pass to the interrupt handler.
54  * @return non-zero if vector is installed.
55  */
56 int caribou_vector_installed(InterruptVector vector,caribou_isr_t isr,void* arg)
57 {
58  int rc =0;
59  int state = caribou_interrupts_disable();
60  if ( vector < (InterruptVector)isr_map_size )
61  {
62  caribou_interrupt_handler_t* node = isr_map[(char)vector];
63  while ( node )
64  {
65  if ( node->isr == isr && node->arg == arg )
66  {
67  rc = 1;
68  break;
69  }
70  node=node->next;
71  }
72  }
73  caribou_interrupts_set(state);
74  return rc;
75 }

77 /**
78  * @brief Install the vector into the isr_map chain.
79  * @param vector The device specific interrupt vector number.
80  * @param isr A pointer to the interrupt service routing for the vector.
81  * @param arg An optional argument to pass to the interrupt handler.
82  * @return The vector number or < 0 on failure.
83  */
84 int caribou_vector_install(InterruptVector vector,caribou_isr_t isr,void* arg)
85 {
86  if ( !caribou_vector_installed(vector,isr,arg) )
87  {
88  int state = caribou_interrupts_disable();
89  if ( isr_map_extend((char)vector+1) > (char)vector )
90  {
91  caribou_interrupt_handler_t* node = (caribou_interrupt_handler_t*)malloc(sizeof(caribou_interrupt_handler_t));
92  if ( node )
93  {
94  memset(node,0,sizeof(caribou_interrupt_handler_t));
95  node->isr = isr;
96  node->arg = arg;
97  if ( isr_map[(char)vector] == NULL )
98  {
99  isr_map[(char)vector] = node;
100  }
101  else
102  {
103  caribou_interrupt_handler_t* next = isr_map[(char)vector];
104  while( next->next ) next = next->next;
105  next->next = node;
106  }
107  }
108  }
109  caribou_interrupts_set(state);
110  return (char)vector;
111  }
112  return -1;
113 }
