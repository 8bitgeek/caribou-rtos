75 /**
76  * @brief Install the vector into the isr_map chain.
77  * @param vector The device specific interrupt vector number.
78  * @param isr A pointer to the interrupt service routing for the vector.
79  * @param arg An optional argument to pass to the interrupt handler.

82 int caribou_vector_install(InterruptVector vector,caribou_isr_t isr,void* arg)
83 {
84  if ( !caribou_vector_installed(vector,isr,arg) )
85  {
86  int state = caribou_interrupts_disable();
87  if ( isr_map_extend((char)vector+1) > (char)vector )
88  {
89  caribou_interrupt_handler_t* node = (caribou_interrupt_handler_t*)malloc(sizeof(caribou_interrupt_handler_t));
90  if ( node )
91  {
92  memset(node,0,sizeof(caribou_interrupt_handler_t));
93  node->isr = isr;
94  node->arg = arg;
95  if ( isr_map[(char)vector] == NULL )
96  {
97  isr_map[(char)vector] = node;
98  }
99  else
100  {

102  while( next->next ) next = next->next;
103  next->next = node;
104  }
105  }
106  }
107  caribou_interrupts_set(state);
