115 /**
116 * @brief Initialize memory.
117 */
118 void bitmap_heap_init(void* heap_base, void* heap_end)
119 {
120  /** Initialize the parameters based on the linkage map */
121  uint32_t heapSegmentSize = (uint32_t)heap_end - (uint32_t)heap_base;
122  uint32_t pageWords = (heapSegmentSize/HEAP_BYTES_PER_PAGE);
123  heap_size = heapSegmentSize - ((pageWords*2)*sizeof(uint32_t));
124  heap_blocks = heap_size / HEAP_BLOCK_SIZE;
125  heap_bitmap_size = heap_blocks / HEAP_BLOCKS_PER_PAGE * sizeof(uint32_t);
126  heap_free_bitmap = (uint32_t*)(heap_base);
127  heap_last_bitmap = (uint32_t*)((uint32_t)heap_base+(pageWords*sizeof(uint32_t)));
128  heap_area = (uint8_t*)((uint32_t)heap_base+((pageWords*2)*sizeof(uint32_t)));
129  memset(heap_area,0,heap_size);
130  memset(heap_free_bitmap,0,heap_bitmap_size);
131  memset(heap_last_bitmap,0,heap_bitmap_size);
132 }

142 /**
143 ** @brief total up all the blocks used.
144 */
145 int16_t bitmap_heap_blocks_allocated()
146 {
147  int16_t n;
148  int16_t total=0;
149  int16_t block=0;
150  while(block < heap_blocks )
151  {
152  uint32_t word = heap_free_bitmap[block_offset(block)];
153  if((word & ALL_BITS) == ALL_BITS)
154  {
155  total += HEAP_BLOCKS_PER_PAGE;
156  }
157  else
158  {
159  for(n=0; n < HEAP_BLOCKS_PER_PAGE; n++)
160  {
161  total += ((word>>n)&1);
162  }
163  }
164  block += HEAP_BLOCKS_PER_PAGE;
165  }
166  return total;
167 }

213 /**
214 * @brief Locate free space
215 * @param blocks The number of free blocks to try and find.
216 * @return The first block.
217 */
218 static int16_t locate_free(int16_t blocks)
219 {
220  int16_t n;
221  int16_t block=0;
222  uint32_t page=0;
223  while(block < heap_blocks)
224  {
225  page = heap_free_bitmap[block_offset(block)];
226  if ( ((page & ALL_BITS) != ALL_BITS) ) /* look at 32 blocks at a time... */
227  {
228  /* out of the next 32 bits, see if we can start the sequence */
229  for(n=0; n < HEAP_BLOCKS_PER_PAGE; n++)
230  {
231  /* try each bit */
232  if (is_free_sequence(block+n,blocks))
233  {
234  return block+n;
235  }
236  }
237  }
238  block += HEAP_BLOCKS_PER_PAGE; /* we've looked at 32 blocks, skip past them */
239  }
240  return -1;
241 }

433 /**
434 ** realloc() changes the size of the memory block pointed to by ptr to size bytes. The contents will be unchanged to the minimum
435 ** of the old and new sizes; newly allocated memory will be uninitialized. If ptr is NULL, then the call is equivalent to malâ
436 ** loc(size), for all values of size; if size is equal to zero, and ptr is not NULL, then the call is equivalent to free(ptr).
437 ** Unless ptr is NULL, it must have been returned by an earlier call to malloc() or realloc(). If the area pointed to
438 ** was moved, a free(ptr) is done.
439 */
440 extern void* bitmap_heap_realloc(void* pointer, size_t size)
441 {
442  if (pointer != NULL && size > 0 )
443  {
444  int16_t blocks = to_blocks(size);
445  int16_t block;
446  int16_t used;
447  int lvl = caribou_lib_lock();
448  block = from_pointer(pointer);
449  if ( block >= 0 )
450  {
451  used = blocks_used(block);
452  if (blocks > used)
453  {
454  if (!extend(block,used,blocks-used)) /* attempt to extend existing block */
455  {
456  int16_t target;
457  deallocate(block,used); /* make currently allocated blocks available to be re-allocated.. */
458  target = locate_free(blocks); /* ...then attempts to locate a sequence of free blocks... */
459  if (target >= 0 )
460  {
461  void* pTarget = allocate(target,blocks); /* allocate the new blocks... */
462  memmove(pTarget,pointer,used*HEAP_BLOCK_SIZE); /* ...and move the data to the new area. */
463  pointer = pTarget;
464  }
465  else
466  {
467  /* The re-allocation failed. The current blocks are already marked free, so just NULL the pointer */
468  notify_heap_invalid_realloc(pointer,size);
469  pointer = NULL;
470  }
471  }
472  }
473  else if (blocks < used)
474  {
475  /* shrink the allocation */
476  deallocate(block,used);
477  allocate(block,blocks);
478  }
479  }
480  else
481  {
482  notify_heap_invalid_realloc(pointer,size);
483  pointer = NULL;
484  }
485  caribou_lib_lock_restore(lvl);
486  }
487  else if (pointer != NULL && size == 0)
488  {
489  bitmap_heap_free(pointer);
490  pointer=NULL;
491  }
492  else if (pointer == NULL )
493  {
494  pointer = bitmap_heap_malloc(size);
495  }
496  return pointer;
497 }
