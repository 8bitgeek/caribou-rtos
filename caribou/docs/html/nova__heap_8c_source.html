49 /**
50 ** malloc() allocates size bytes and returns a pointer to the allocated memory. The memory is not cleared. If size is 0, then
51 ** malloc() returns either NULL, or a unique pointer value that can later be successfully passed to free().
52 */
53 void* nova_heap_malloc(size_t size)
54 {
55  heap_t* p;
56  heap_t* q;
57  heap_t* best;
58  uint32_t best_size;
59 
60  int lvl = caribou_lib_lock();
61 
62  size = (size + sizeof(uint32_t) + HEAP_BLOCK_SIZE-1) & (uint32_t)(-HEAP_BLOCK_SIZE);
63  p = heap;
64  q = 0;
65  best_size = ~0U;
66  best = 0;
67 
68  while (p)
69  {
70  if (p->size == size) // done?
71  {
72  if (q)
73  {
74  q->next = p->next;
75  }
76  else
77  {
78  heap = p->next;
79  }
80  *((uint32_t *)p) = size;
81  caribou_lib_lock_restore(lvl);
82  return ((uint8_t *)p) + sizeof(uint32_t);
83  }
84  else
85  if (p->size > size && p->size < best_size) // split block?
86  {
87  best = p;
88  best_size = p->size;
89  }
90  /* Consider next block. */
91  q = p;
92  p = p->next;
93  }
94  if (best) // split
95  {
96  best->size -= size;
97  p = ADDADR(best, best_size - size);
98  *((uint32_t *)p) = size;
99  caribou_lib_lock_restore(lvl);
100  return ((char *)p) + sizeof(uint32_t);
101  }
102  else
103  {
104  caribou_lib_lock_restore(lvl);

106  }
107 }

109 /**
110 ** free() frees the memory space pointed to by ptr, which must have been returned by a previous call to malloc() or
111 ** realloc(). Otherwise, or if free(ptr) has already been called before, undefined behavior occurs. If ptr is NULL, no opera‐
112 ** tion is performed.
113 */
114 void nova_heap_free(void *addr)
115 {
116  heap_t *p, *rover, *prev;
117  unsigned size;
118 
119  if ( addr )
120  {
121  int lvl = caribou_lib_lock();
122  /* Adjust backwards to actual memory block origin. */
123  addr = ((char *)addr) - sizeof(unsigned);
124  /* Recover size */
125  size = *((unsigned *)addr);
126  if (addr)
127  {
128  /* Deallocated onto non-existent heap? */
129  if (heap == 0)
130  {
131  /* Yes, add this floating block to the heap. */
132  heap = addr;
133  heap->next = 0;
134  heap->size = size;
135  }
136  else
137  {
138  /* Search for block adjacent to p, but with lower address. */
139  p = addr; rover = heap; prev = 0;
140  while (rover && rover < (heap_t *)addr)
141  {
142  prev = rover; rover = rover->next;
143  }
144  /* Check for double-dispose of a pointer. */
145  /* For efficiency, this could be removed. */
146  #if 0
147  if ((rover = addr) ||
148  prev & (DIFADR(addr, ADDADR(prev, prev->size)) < 0))
149  HALT(DOUBLEDISPOSE);
150  END;
151  #endif
152  /* Return block to free list */
153  p->next = rover;
154  p->size = size;
155  if (prev)
156  prev->next = p;
157  else
158  heap = p;
159  /* See if can join with block to right. */
160  if (rover && (ADDADR(p, size) == rover))
161  {
162  p->size += rover->size;
163  p->next = rover->next;
164  }
165  /* See if can join with block to left. */
166  if (prev && ADDADR(prev, prev->size) == p)
167  {
168  prev->size += p->size;
169  prev->next = p->next;
170  }
171  }
172  }
173  caribou_lib_lock_restore(lvl);
174  }
175 }
