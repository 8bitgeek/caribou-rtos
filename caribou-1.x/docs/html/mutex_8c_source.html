66 /// Try to acquire lock. If the mutex is currently held, the calling thread
67 /// will block unless the current thread is the holder of the current lock.
68 /// @param mutex A pointer to the caribou_mutex_t struct.
69 /// @param timeout The timeout to wait in milliseconds, or forever if zero.
70 /// @return true if the lock was acquired.
71 bool caribou_mutex_lock(caribou_mutex_t* mutex,uint32_t timeout)
72 {
73  caribou_timer_t timer;
74  int state = caribou_lock();
75  ++mutex->blocking;
76  caribou_lock_set(state);
77  if ( timeout )
78  {
79  caribou_timer_set(&timer,from_ms(timeout));
80  caribou_timer_init(&timer,NULL,NULL,CARIBOU_TIMER_F_ONESHOT);
81  }
82  while ( !caribou_mutex_trylock(mutex) && (!timeout || (timeout && !caribou_timer_expired(&timer))))
83  {
84  if ( (mutex->flags & CARIBOU_MUTEX_F_NOYIELD) == 0 )
85  {
86  caribou_thread_yield();
87  }
88  }
89  state = caribou_lock();
90  --mutex->blocking;
91  caribou_lock_set(state);
92  return true;
93 }

95 /// Try to acquire a mutex lock. If the mutex is currently held, the calling thread
96 /// will not block, but caribou_mutex_trylock() will return false.
97 /// @param A pointer to the caribou_mutex_t struct.
98 /// @return true if the lock was acquired.
99 bool caribou_mutex_trylock(caribou_mutex_t* mutex)
100 {
101  bool rc;
102  caribou_thread_t* current_thread = caribou_thread_current();
103  int state = caribou_lock();
104  if ( mutex->locks && mutex->thread != current_thread )
105  {
106  /* not acquired */
107  rc = false;
108  }
109  else
110  {
111  /* acquired */
112  ++mutex->locks; // increment number of locks
113  mutex->thread = current_thread;
114  rc = true;
115  }
116  caribou_lock_set(state);
117  return rc;
118 }

120 /// @brief Unlock a mutex previously acquired with thread_mutex_lock() or thread_caribou_mutex_trylock()
121 /// If there are theads blocking on the mutex, then yield after releasing the mutex, otherwise
122 /// if there are no other threads blocking on the mutex, then continue without yielding.
123 bool caribou_mutex_unlock(caribou_mutex_t* mutex)
124 {
125  bool rc=false;
126  int state = caribou_lock();
127  if ( mutex->locks && mutex->thread == caribou_thread_current() )
128  {
129  if ( --mutex->locks == 0 ) // decrement number of locks.
130  mutex->thread=NULL;
131  rc=true;
132  }
133  // if there are blockers, then yield, otherwise not...
134  if ( mutex->blocking )
135  {
136  caribou_lock_set(state);
137  if ( (mutex->flags & CARIBOU_MUTEX_F_NOYIELD) == 0 )
138  {
139  caribou_thread_yield();
140  }
141  }
142  else
143  {
144  caribou_lock_set(state);
145  }
146  return rc;
147 }
