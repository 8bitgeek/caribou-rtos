408 /**
409  * Create a new instance of a CARIBOU thread.
410  * @param name The ascii name of the thread. This pointer must remain valid for the life span
411  * of the thread instance.
412  * @param run A Pointer to the entry point function of the thread. Note the entry point function must accept
413  * a void* pointer even if it is not used.
414  * @param finish A function which is called up thread termination.
415  * @param arg An argument pointer which is passed to the thread entry point function run.
416  * @param stackaddr A pointer to bottom of the program stack associated with this thread.
417  * @param stack_size The size of the stack belonging to this thread.
418  * @param priority The priority of the thread.
419  * @return A pointer to the newly created thread or NULL if something failed.
420  */
421 caribou_thread_t* caribou_thread_create(const char* name, void (*run)(void*), void (*finish)(void*), void * arg, void * stackaddr, int stack_size, int16_t priority )
422 {
423  caribou_thread_t* node=NULL;
424  process_frame_t* process_frame;
425  int state = chip_interrupts_disable();
426 
427  if ( stackaddr )
428  {
429  //initialize the process stack pointer
430  memset(stackaddr,0xFA,stack_size);
431  process_frame = (hw_stack_frame_t *)(stackaddr + stack_size - sizeof(process_frame_t) );
432  memset(process_frame,0,sizeof(process_frame_t));
433  process_frame->hw_stack.r0 = (uint32_t)arg;
434  process_frame->hw_stack.lr = (uint32_t)thread_finish;
435  process_frame->hw_stack.pc = (((uint32_t)run) + INITIAL_PC_OFFSET) & (~1);
436  process_frame->hw_stack.psr = DEFAULT_PSR;
437  #if defined(ARM_FVP_LAZY_STACKING)
438  process_frame->sw_stack.lr = DEFAULT_EXCEPTION_RETURN;
439  #endif
440  }
441 
442  node = new_thread_node(_current_thread_==NULL?_caribou_thread_queue_:_current_thread_);
443  if ( node != NULL )
444  {
445  node->state = 0;
446  if ( stackaddr )
447  {
448  uint32_t stack_top = (stackaddr + stack_size);
449  node->sp = stack_top - sizeof(process_frame_t);
450  node->stack_top = stackaddr + stack_size;
451  node->stack_low = stackaddr;
452  node->stack_low += sizeof(process_frame_t);
453  node->stack_base = stackaddr;
454  }
455  else
456  {
457  node->sp = (&__process_stack_end__);
458  node->stack_top = node->sp;
459  node->stack_low = (&__process_stack_base__);
460  node->stack_low += sizeof(process_frame_t);
461  node->stack_base = (&__process_stack_base__);
462  }
463  node->name = name;
464  node->arg = arg;
465  node->prio = priority;
466  node->finishfn = finish;
467  append_thread_node(node);
468  }
469  chip_interrupts_set(state);
470  return node;
471 }

586 static void check_sp(caribou_thread_t* thread)
587 {
588  if ( thread->sp <= thread->stack_low || thread->sp > thread->stack_top )
589  {
590  if ( thread->sp <= thread->stack_low )
591  {
592  _caribou_thread_fault_emit(THREAD_FAULT_STACK_LOW);
593  }
594  else if ( thread->sp <= thread->stack_base )
595  {
596  _caribou_thread_fault_emit(THREAD_FAULT_STACK_OVERFLOW);
597  }
598  else if ( thread->sp > thread->stack_top )
599  {
600  _caribou_thread_fault_emit(THREAD_FAULT_STACK_UNDERFLOW);
601  }
602  }
603  if ( thread->sp < thread->stack_usage || !thread->stack_usage )
604  {
605  thread->stack_usage = thread->sp;
606  }
607 }

609 void caribou_thread_once()
610 {
611  caribou_thread_t* thread=_caribou_thread_queue_;
612  caribou_thread_t* next;
613 
614  // idle hooks...
615  board_idle();
616  // idle the timers...
617  caribou_timer_idle(thread);
618  // terminate threads flagged for termination...
619  for(; thread!=NULL; thread=next)
620  {
621  next = thread->next; /* capture the next in case we terminate this thread */
622  check_sp(thread);
623  if ( thread->state & CARIBOU_THREAD_F_TERMINATED && thread != _current_thread_ )
624  {
625  caribou_thread_terminate(thread);
626  }
627  }
628 }

644 /**
645  * @brief Swap context to a the next thread in the queue.
646  */
647 static void _swap_thread()
648 {
649  _current_thread_->errno = errno; // store the thread's errno
650  if ( preemptable( _current_thread_ ) && priority-- < 0 )
651  {
652  while( !runnable((_current_thread_=nextinqueue(_current_thread_))) )
653  {
654  _current_thread_->state &= ~CARIBOU_THREAD_F_YIELD;
655  }
656  if ( ( priority = _current_thread_->prio ) < 0)
657  {
658  priority = 0;
659  }
660  errno = _current_thread_->errno; // restore the thread's errno
661  }
662 }
