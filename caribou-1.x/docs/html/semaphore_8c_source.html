83 /** **************************************************************************
84  ** @brief This operation increases the semaphore counter, if the result
85  ** is non-negative then a waiting thread is removed from the
86  ** queue and resumed. Threads are removed from the queue in a FIFO manner.
87  ** @param semaphore The semaphore to operate on.
88  ** @return true if the semaphore was released.
89  *****************************************************************************/
90 bool caribou_semaphore_signal(caribou_semaphore_t* semaphore)
91 {
92  bool rc=true;
93  int state = caribou_lib_lock();
94  if ( ++semaphore->count <= 0 )
95  {
96  caribou_thread_t* thread;
97  if ( semaphore->queue && (thread = caribou_queue_take_first(semaphore->queue,TIMEOUT_IMMEDIATE)) )
98  {
99  caribou_thread_wakeup(thread);
100  }
101  else
102  {
103  --semaphore->count;
104  rc=false;
105  }
106  }
107  caribou_lib_lock_restore(state);
108  return rc;
109 }

111 /** **************************************************************************
112  ** @brief This operation decreases the semaphore counter, if the result
113  ** is negative then the invoking thread is queued.
114  ** @param semaphore The semaphore to operate on.
115  ** @return true of the semaphore was released.
116  *****************************************************************************/
117 bool caribou_semaphore_wait(caribou_semaphore_t* semaphore, caribou_tick_t timeout)
118 {
119  bool rc;
120  caribou_tick_t start = caribou_timer_ticks();
121  while (!(rc=caribou_semaphore_try_wait(semaphore)) && !caribou_timer_ticks_timeout(start,timeout) )
122  caribou_thread_yield();
123  return rc;
124 }

126 /** **************************************************************************
127  ** @brief This operation decreases the semaphore counter, if the result
128  ** is negative then the invoking thread is queued.
129  ** @param semaphore The semaphore to operate on.
130  ** @return true of the semaphore was released.
131  *****************************************************************************/
132 bool caribou_semaphore_try_wait(caribou_semaphore_t* semaphore)
133 {
134  bool rc=true;
135  int state = caribou_lib_lock();
136  if ( --semaphore->count < 0 )
137  {
138  if ( semaphore->queue && caribou_queue_try_post(semaphore->queue,caribou_thread_current()) )
139  {
140  caribou_lib_unlock();
141  caribou_thread_sleep(caribou_thread_current(),TIMEOUT_INFINITE);
142  caribou_lib_lock();
143  }
144  else
145  {
146  ++semaphore->count; /* failed to wait */
147  rc=false;
148  }
149  }
150  caribou_lib_lock_restore(state);
151  return rc;
152 }
