32  " subs %2,%2,#1 \n" /* decrement count and set/reset Z flag */
33  " b __memcpy_1 \n" /* next... */
34  "__memcpy_2: \n"

425 int atoi(const char* a)
426 {

429  {

433  ++digit; // skip to last digit
434  for(--digit; digit >= 0 && ( isnum(a[digit]) || a[digit] == '.' || a[digit]=='-' ); digit-- )
435  {
436  if ( a[digit] == '-' )
437  {
438  rc = -rc;

440  }

442  {
443  multiplier=1;
444  rc=0;
445  }
446  else
447  {
448  rc += multiplier * (a[digit] - '0');
449  multiplier *= 10;
450  }
451  }
452  }
