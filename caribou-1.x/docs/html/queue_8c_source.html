25 /** **************************************************************************
26  ** @brief Allocate and initialize a new queue on the stack
27  ** @param depth The depth of the queue expressed in the number of pointers.
28  ** @return A pointer to the new queue or NULL if allocation failed.
29  *****************************************************************************/
30 caribou_queue_t* caribou_queue_new(int depth)
31 {
32  caribou_queue_t* queue=(caribou_queue_t*)malloc(sizeof(caribou_queue_t));
33  if(queue)
34  {
35  if ( depth == QUEUE_DEPTH_DYNAMIC )
36  {
37  queue->msgs=(caribou_queue_msg_t**)NULL;
38  queue->depth = depth;
39  queue->count=0;
40  }
41  else
42  {
43  queue->msgs=(caribou_queue_msg_t**)malloc(depth*sizeof(caribou_queue_msg_t*));
44  if ( queue->msgs )
45  {
46  memset(queue->msgs,0,depth*sizeof(caribou_queue_msg_t*));
47  queue->depth = depth;
48  queue->count=0;
49  }
50  else
51  {
52  free(queue);
53  queue=NULL;
54  }
55  }
56  }
57  return queue;
58 }

216 /** **************************************************************************
217  ** @brief Post a message to the queue and don't block if there is no space in
218  ** the queue.
219  ** @param queue The queue to operate on.
220  ** @param msg he message to insert.
221  ** @return true if message was posted
222  *****************************************************************************/
223 bool caribou_queue_try_post(caribou_queue_t* queue, caribou_queue_msg_t* msg)
224 {
225  bool rc=false;
226  int lvl = caribou_queue_lock();
227  if ( queue->depth == QUEUE_DEPTH_DYNAMIC ) // dynamic queue?
228  {
229  queue->msgs = (caribou_queue_msg_t**)realloc(queue->msgs,(queue->count+1)*sizeof(caribou_queue_msg_t*));
230  if ( queue->msgs )
231  {
232  queue->msgs[queue->count++] = msg;
233  rc=true;
234  }
235  }
236  else if ( queue->count < queue->depth ) // static queue has space?
237  {
238  queue->msgs[queue->count++] = msg;
239  rc=true;
240  }
241  caribou_queue_lock_restore(lvl);
242  return rc;
243 
244 }

312 {
