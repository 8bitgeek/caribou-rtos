51 /**
52 ** malloc() allocates size bytes and returns a pointer to the allocated memory. The memory is not cleared. If size is 0, then
53 ** malloc() returns either NULL, or a unique pointer value that can later be successfully passed to free().
54 */
55 void* nova_heap_malloc(size_t size)
56 {
57  heap_t* p;
58  heap_t* q;
59  heap_t* best;
60  uint32_t best_size;
61 
62  int lvl = caribou_lib_lock();
63 
64  size = (size + sizeof(uint32_t) + HEAP_BLOCK_SIZE-1) & (uint32_t)(-HEAP_BLOCK_SIZE);
65  p = heap;
66  q = 0;
67  best_size = ~0U;
68  best = 0;
69 
70  while (p)
71  {
72  if (p->size == size) // done?
73  {
74  if (q)
75  {
76  q->next = p->next;
77  }
78  else
79  {
80  heap = p->next;
81  }
82  *((uint32_t *)p) = size;
83  caribou_lib_lock_restore(lvl);
84  return ((uint8_t *)p) + sizeof(uint32_t);
85  }
86  else
87  if (p->size > size && p->size < best_size) // split block?
88  {
89  best = p;
90  best_size = p->size;
91  }
92  /* Consider next block. */
93  q = p;
94  p = p->next;
95  }
96  if (best) // split
97  {
98  best->size -= size;
99  p = ADDADR(best, best_size - size);
100  *((uint32_t *)p) = size;
101  caribou_lib_lock_restore(lvl);
102  return ((char *)p) + sizeof(uint32_t);
103  }
104  else
105  {
106  caribou_lib_lock_restore(lvl);
107  return NULL;
108  }
109 }

114 ** tion is performed.

116 void nova_heap_free(void *addr)
117 {

119  unsigned size;
120 
121  if ( addr )
122  {
123  int lvl = caribou_lib_lock();
124  /* Adjust backwards to actual memory block origin. */
125  addr = ((char *)addr) - sizeof(unsigned);
126  /* Recover size */
127  size = *((unsigned *)addr);
128  if (addr)
129  {
130  /* Deallocated onto non-existent heap? */
131  if (heap == 0)
132  {
133  /* Yes, add this floating block to the heap. */
134  heap = addr;
135  heap->next = 0;
136  heap->size = size;
137  }
138  else
139  {
140  /* Search for block adjacent to p, but with lower address. */
141  p = addr; rover = heap; prev = 0;
142  while (rover && rover < (heap_t *)addr)
143  {
144  prev = rover; rover = rover->next;
145  }
146  /* Check for double-dispose of a pointer. */
147  /* For efficiency, this could be removed. */
148  #if 0
149  if ((rover = addr) ||
150  prev & (DIFADR(addr, ADDADR(prev, prev->size)) < 0))
151  HALT(DOUBLEDISPOSE);
152  END;
153  #endif
154  /* Return block to free list */
155  p->next = rover;
156  p->size = size;
157  if (prev)
158  prev->next = p;
159  else
160  heap = p;
161  /* See if can join with block to right. */
162  if (rover && (ADDADR(p, size) == rover))
163  {
164  p->size += rover->size;
165  p->next = rover->next;
166  }
167  /* See if can join with block to left. */
168  if (prev && ADDADR(prev, prev->size) == p)
169  {
170  prev->size += p->size;
171  prev->next = p->next;
172  }
173  }
174  }
175  caribou_lib_lock_restore(lvl);
176  }
177 }
