39 /**
40  * @brief Open a UART device for subsequent use.
41  * @param devicenum Specifies the device number to use.
42  * @return The file descriptor or < 0 on error.
43  */
44 int caribou_uart_open(int ndev,caribou_uart_config_t* config)
45 {
46  int fd;
47  errno = EOKAY;
48  caribou_thread_lock();
49  if ( (fd = ndev) >= 0 )
50  {
51  stdio_t* io = &_stdio_[fd];
52  void* device = io->device_private;
53  caribou_bytequeue_free(chip_uart_rx_queue(device));
54  caribou_bytequeue_free(chip_uart_tx_queue(device));
55  chip_uart_set_rx_queue(device,NULL);
56  chip_uart_set_tx_queue(device,NULL);
57  if ( chip_uart_set_rx_queue(device,caribou_bytequeue_new(caribou_uart_queue_rx_sz())) )
58  {
59  if ( chip_uart_set_tx_queue(device,caribou_bytequeue_new(caribou_uart_queue_tx_sz())) )
60  {
61  chip_uart_set_config(device,NULL); /* set defaults */
62  chip_uart_set_status(device,chip_uart_status(device) | STDIO_STATE_OPENED);
63  }
64  else
65  {
66  caribou_bytequeue_free(chip_uart_rx_queue(device));
67  errno = ENOMEM;
68  }
69  }
70  else
71  {
72  errno = ENOMEM;
73  }
74  }
75  caribou_thread_unlock();
76  if ( fd >= 0 && config )
77  {
78  if ( caribou_uart_set_config(fd,config) < 0 )
79  {
80  caribou_uart_close(fd);
81  fd = (-1);
82  }
83  }
84  if ( chip_uart_open(fd) != fd )
85  {
86  caribou_uart_close(fd);
87  }
88  return fd;
89 }

91 /**
92  * @brief Close a previously opened UART.
93  * @param fd The opened UART file descriptor.
94  * @return < 0 on error.
95  */
96 int caribou_uart_close(int fd)
97 {
98  int rc=0;
99  errno = EOKAY;
100  caribou_thread_lock();
101  stdio_t* io = &_stdio_[fd];
102  void* device = io->device_private;
103  chip_uart_int_disable(device);
104  chip_vector_disable(chip_uart_interrupt_vector(device));
105  caribou_bytequeue_free(chip_uart_rx_queue(device));
106  caribou_bytequeue_free(chip_uart_tx_queue(device));
107  chip_uart_set_rx_queue(device,NULL);
108  chip_uart_set_tx_queue(device,NULL);
109  chip_uart_set_status(device,chip_uart_status(device) & ~STDIO_STATE_OPENED);
110  caribou_thread_unlock();
111  return rc;
112 }

262 /// Device Driver read-data function.
263 /// @return number of bytes read, or < 0 + errno
264 int caribou_uart_private_readfn(stdio_t* io,void* data,int count)
265 {
266  int rc=0;
267  uint8_t* p = (uint8_t*)data;
268  while (rc < count)
269  {
270  int c;
271  if ( (c = caribou_bytequeue_get(chip_uart_rx_queue(io->device_private))) >= 0 )
272  {
273  p[rc++] = (uint8_t)(c&0xff);
274  chip_uart_set_status(io->device_private,chip_uart_status(io->device_private) | STDIO_STATE_RX_PENDING);
275  }
276  else
277  {
278  caribou_thread_yield();
279  }
280  }
281  return rc;
282 }

284 /// Device Driver write-data function.
285 int caribou_uart_private_writefn(stdio_t* io,void* data,int count)
286 {
287  int rc=0;
288  uint8_t* p = (uint8_t*)data;
289  while( rc < count )
290  {
291  // Fill the transmitter queue as much as we can...
292  if ( caribou_bytequeue_put(chip_uart_tx_queue(io->device_private),p[rc]) )
293  {
294  ++rc;
295  }
296  else
297  {
298  caribou_thread_yield();
299  }
300  // Allow the interrupt service routine to empty out the tx queue a little...
301  chip_uart_tx_start(io->device_private);
302  }
303  return rc;
304 
305 }
