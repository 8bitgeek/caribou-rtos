56  /// @brief Get the next socket in the queue ala round-robin
57  CARIBOU::CUdpSocket& CUdpSession::nextSocket()
58  {
59  lock();
60  if ( socketQueue().count() )
61  {
62  // if the current socket is valid, then find the next in the queue
63  // or retain the existing socket if it's the only one in the queue.
64  if ( socketQueue().count() > 1 )
65  {
66  int idx = socketQueue().indexOf(mSocket.socket())+1;
67  if ( idx >= socketQueue().count() )
68  idx=0;
69  setSocket(socketQueue().at(idx));
70  }
71  else
72  {
73  setSocket(socketQueue().at(0));
74  }
75  }
76  unlock();
77  return mSocket;
78  }
