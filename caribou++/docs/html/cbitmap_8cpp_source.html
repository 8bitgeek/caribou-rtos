76  void CBitmap::flip(register uint8_t* output, register const uint8_t* input, int size)
77  {
78  register uint32_t inch;
79  register uint32_t outch;
80  for(int byte=0; byte < size; byte++)
81  {
82  inch = *input++;
83  outch = 0;
84  for(int bit=0; bit < 8; bit++)
85  {
86  outch >>= 1;
87  outch |= (inch&0x80);
88  inch <<= 1;
89  }
90  *output++ = (uint8_t)outch;
91  }
