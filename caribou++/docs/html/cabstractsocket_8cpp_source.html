288  int CAbstractSocket::read(CARIBOU::CByteArray& buf, int len, int timeout)
289  {
290  char t[64];
291  int rc=0;
292  char ch;
293  caribou_tick_t start = (uint32_t)from_ms(caribou_timer_ticks());
294  buf.clear();
295  while ( !caribou_timer_ticks_timeout(start,from_ms(timeout)) && (len == 0 || buf.length() < len) )
296  {
297  int nBytes = lwip_recv(mSocket,t,64,MSG_DONTWAIT);
298  if ( nBytes > 0 )
299  {
300  if ( len == 0 )
301  {
302  char* dst;
303  int pos = buf.length();
304  buf.resize(buf.length()+nBytes);
305  dst = &(buf.data()[pos]);
306  memcpy(dst,t,nBytes);
307  }
308  else
309  {
310  char* dst;
311  int pos = buf.length();
312  int nRemaining = len - buf.length();
313  int nRead = nBytes > nRemaining ? nRemaining : nBytes;
314  buf.resize(buf.length()+nRead);
315  dst = &(buf.data()[pos]);
316  memcpy(dst,t,nBytes);
317  }
318  }
319  else if ( len == 0 && buf.length() )
320  {
321  // undetermined max length, we already read some data, and none is available, so we're done..
322  break;
323  }
324  else
325  {
326  caribou_thread_yield();
327  }
328  }
329  return rc < 0 ? rc : buf.length();
330  }
