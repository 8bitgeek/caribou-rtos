878  clear();
879

881  {
882  long_long = false;
883  if ( *p == '%' )
884  {
885  ++p;
886 
887  if ( *p == '0' )
888  {
889  zeros = true;
890  ++p;
891  }

894  zeros = false;
895  }

899  {

904

906  {

908  ++p;
909  }
910 
911  switch (*p )
912  {
913  case 'X':
914  case 'x':
915  case 'd':
916  case 'u':

937  break;
938  }
939  }

943  {
944  default:
945  case 0:
946  dec.toHex( (uint32_t)va_arg(ap,uint32_t) );
947  break;
948  case 2:
949  dec.toHex( (uint8_t)va_arg(ap,uint32_t) );
950  break;
951  case 4:
952  dec.toHex( (uint16_t)va_arg(ap,uint32_t) );
953  break;
954  case 8:
955  dec.toHex( (uint32_t)va_arg(ap,uint32_t) );
956  break;
957  }
958  }
959  }
960  else if (*p == 'd' )
961  {
962  if ( long_long )

971  else
972  dec.fromUBit32( (int32_t)va_arg(ap,int32_t) );
973  }
974  while ( dec.length() < fill )
975  {
976  dec.prepend( zeros ? '0' : ' ' );
977  }
978  append( dec );
979  }
980  break;
981  case 'c':
982  append( (char)va_arg(ap,int) );
983  break;
984  case 's':
985  {
986  char* s = (char*)va_arg(ap,char*);
987  if ( !s )
988  s="(null)";
989  append( (char*)s );
990  }
991  break;
