276  void CObject::dispatch(CEvent* e)
277  {
278  for(int n=0; n < mListenerMap.count(); n++)
279  {
280  CARIBOU::CEvent::Type listeningType = (CARIBOU::CEvent::Type)mListenerMap.dataAt(n);
281  if ( listeningType == e->type() )
282  {
283  CObject* obj = mListenerMap.at(n);
284  if ( e->receiver() == NULL || (e->receiver() != NULL && obj == e->receiver()) )
285  {
286  //syslog(0,true,"dispatch> [%08X][%08X][%08X]",obj,e->sender(),e->receiver());
287  obj->event(e);
288  //syslog(0,true,"dispatch< [%08X][%08X][%08X]",obj,e->sender(),e->receiver());
289  }
290  }
291  }
293  if ( !e->senderOwns() )
294  {
295  delete e;
296  }
297  }

302  int CObject::purge(CObject* object)
303  {
304  int rc=0;
305  objectLock();
306  /* @brief If the object is referenced in the listener map, then remove each reference... */
307  int idx;
308  while ( (idx = mListenerMap.indexOf(object)) >= 0 )
309  {
310  mListenerMap.take(idx);
311  }
316  for ( int n=0; n < mEventQueue.count(); n++ )
317  {
318  CEvent* e = mEventQueue.at(n);
319  if ( e )
320  {
321  if ( e->sender() == object || e->receiver() == object )
322  {
323  mEventQueue.take(n);
324  delete e;
325  }
326  }
327  }
328  objectUnlock();
329  return rc;
