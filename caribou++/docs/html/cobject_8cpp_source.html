242  /**
243  * Enqueue an event for later dispatch.
244  * CEventQueue has event ownership and will delete it unless e->senderOwns() is true.
245  * @return true if the event was queued.
246  */
247  bool CObject::enqueue(CEvent* e)
248  {
249  bool rc=false; /** Only enqueue the event if somebody is listening to it's type... */
250  //syslog(0,true,"enqueue> [%08X][%08X]",e->sender(),e->receiver());
251  objectLock();
252  if ( ((CEvent*)e)->priority() == CEvent::PriorityHigh )
253  {
254  /** jump the queue - LIFO mode */
255  /** TODO - load testing required. test threshold where flooding hi priority events begins to starve the queue. */
256  mEventQueue.insert(e,0);
257  }
258  else
259  {
260  /** normal/low priority - FIFO mode. */
261  mEventQueue.append(e);
262  }
263  rc=true;
264  objectUnlock();
265  //syslog(0,true,"enqueue< [%08X][%08X]",e->sender(),e->receiver());
266  return rc;
267  }

269  /**
270  * @brief dispatch a particular event registered listener object's event(...) method.
271  * @param e The event to dispatch must already be detached from the event queue.
272  * @note dispatch(...) owns the pointer.
273  * @note Thread locks are used to guard the listener map. This implies that the listening object
274  * should not use a MUTEX while receiving the event in order to avoid a deadlock condition.
275  */
276  void CObject::dispatch(CEvent* e)
277  {
278  for(int n=0; n < mListenerMap.count(); n++)
279  {
280  CARIBOU::CEvent::Type listeningType = (CARIBOU::CEvent::Type)mListenerMap.dataAt(n);
281  if ( listeningType == e->type() )
282  {
283  CObject* obj = mListenerMap.at(n);
284  if ( e->receiver() == NULL || (e->receiver() != NULL && obj == e->receiver()) )
285  {
286  //syslog(0,true,"dispatch> [%08X][%08X][%08X]",obj,e->sender(),e->receiver());
287  obj->event(e);
288  //syslog(0,true,"dispatch< [%08X][%08X][%08X]",obj,e->sender(),e->receiver());
289  }
290  }
291  }
292  /** Then we delete the event object... */
293  if ( !e->senderOwns() )
294  {
295  delete e;
296  }
297  }

299  /**
300  * @brief purge all references to the specified object from the listener map.
301  */
302  int CObject::purge(CObject* object)
303  {
304  int rc=0;
305  objectLock();
306  /* @brief If the object is referenced in the listener map, then remove each reference... */
307  int idx;
308  while ( (idx = mListenerMap.indexOf(object)) >= 0 )
309  {
310  mListenerMap.take(idx);
311  }
312  /**
313  * @brief If there are any events in the queue that are addressed to or from the object,
314  * then remove them...
315  */
316  for ( int n=0; n < mEventQueue.count(); n++ )
317  {
318  CEvent* e = mEventQueue.at(n);
319  if ( e )
320  {
321  if ( e->sender() == object || e->receiver() == object )
322  {
323  mEventQueue.take(n);
324  delete e;
325  }
326  }
327  }
328  objectUnlock();
329  return rc;
330  }
